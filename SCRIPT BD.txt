CREATE TABLE BAIRRO(
	BAIRRO_ID 	INTEGER PRIMARY KEY,
	BAIRRO_NOME 	VARCHAR(20) NOT NULL,
	BAIRRO_TAXA 	REAL
);

CREATE TABLE FORM_PAG(
	PAG_ID 	  	INTEGER PRIMARY KEY,
	PAG_DESCRICAO   VARCHAR(20) NOT NULL
);

CREATE TABLE FUNCIONARIOS(
	FUNC_ID 	INTEGER PRIMARY KEY,
	FUNC_NOME 	VARCHAR(50) NOT NULL,
	FUNC_SENHA 	VARCHAR(20) NOT NULL,
	FUNC_ACESSO 	VARCHAR(10) NOT NULL
);

CREATE TABLE CATEGORIA(
	CATEG_ID 	INTEGER PRIMARY KEY,
	CATEG_DESCRICAO VARCHAR(50) NOT NULL
);

CREATE TABLE CARDAPIO(
	CARD_ID 	INTEGER PRIMARY KEY,
	CARD_NOME 	VARCHAR(50) NOT NULL,
	CARD_DESCRICAO  VARCHAR(500) NOT NULL,
	CARD_FOTO 	VARCHAR(150) NOT NULL,
	CATEG_ID_FK 	INTEGER,
	CARD_VALOR 	NUMERIC(3,2) NOT NULL,
	FOREIGN KEY (CATEG_ID_FK) REFERENCES CATEGORIA(CATEG_ID)
);

ALTER TABLE BAIRRO ADD COLUMN UF CHAR(2) NOT NULL;

CREATE TABLE CLIENTE(
	CLIENTE_CPF 		CHAR(14) PRIMARY KEY,
	CLIETNE_NOME 		VARCHAR(60) NOT NULL,
	CLIENTE_TELEFONE 	VARCHAR(14) NOT NULL,
	CLIENTE_ENDERECO 	VARCHAR(50) NOT NULL,
	CLIENTE_COMPLEMENTO 	VARCHAR(50),
	CLIENTE_NUM_END 	INTEGER NOT NULL,
	CLIENTE_CEP 		CHAR(10) NOT NULL,
	CLIENTE_ID_BAIRRO 	INTEGER NOT NULL,
	FOREIGN KEY (CLIENTE_ID_BAIRRO) REFERENCES BAIRRO(BAIRRO_ID)
);

CREATE TABLE PEDIDOS(
	PEDIDOS_ID 		INTEGER PRIMARY KEY,
	PEDIDOS_QTDE 		INTEGER NOT NULL,
	PEDIDOS_VALOR 		NUMERIC(15,9) NOT NULL,
	PEDIDOS_DATA 		DATE NOT NULL,
	PEDIDOS_HORA 		TIME WHITHOUT TIME ZONE NOT NULL,
	PEDIDOS_STATUS 		VARCHAR(20) NOT NULL,
	PEDIDOS_TROCO 		NUMERIC(15,9),
	PEDIDOS_ID_BAIRRO_FK 	INTEGER NOT NULL,
	PEDIDOS_CPF_CLIETNE_FK 	CHAR(14) NOT NULL,
	
	PEDIDOS_ID_FORM_PGT 	INTEGER NOT NULL,
	FOREIGN KEY (PEDIDOS_ID_BAIRRO_FK) REFERENCES BAIRRO (BAIRRO_ID),
	FOREIGN KEY (PEDIDOS_CPF_CLIETNE_FK) REFERENCES CLIENTE (CLIENTE_CPF),
	
	FOREIGN KEY (PEDIDOS_ID_FORM_PGT) REFERENCES FORM_PAG (PAG_ID)
);

CREATE TABLE VENDAS(
	VENDAS_ID 		INTEGER PRIMARY KEY,
	VENDAS_DATA 		DATE NOT NULL,
	VENDAS_HORA 		TIME NOT NULL,
	VENDAS_ID_PEDIDOS_FK 	INTEGER NOT NULL,
	FOREIGN KEY (VENDAS_ID_PEDIDOS_FK) REFERENCES PEDIDOS (PEDIDOS_ID)
);

CREATE TABLE card_ped(
id_card_ped integer not null,
id_card integer NOT NULL,
id_ped integer NOT NULL,
quant integer,
FOREIGN KEY (id_card) REFERENCES CARDAPIO (CARD_ID),
FOREIGN KEY (id_ped) REFERENCES PEDIDOS (pedidos_id)
);
